    public deprecate(pred: Boolean, format: String, args: ...args<any>): void source 

    import {deprecate} from 'groundwork.js/src/utils.js'

    Log a deprecation warning in console only in DEV mode

    Params:

    NameTypeAttributeDescription
    pred Boolean

    predicate or value should be "truthy" or "falsey"
    format String

    warning message template
    args ...args<any>

    values to substitue into format

    Return:

    void

    Example:

    const opts = {bar: 1};
deprecate(opts.bar, 'opts.bar is deprecated!') // 'opts.bar is deprecated'

const foo = 1;
deprecate((typeof foo === String), 'String support is deprecated, given %s', foo)
// 'String support is deprecated, given 1'

    public has(object: Object, path: Array | string): boolean source 

    import {has} from 'groundwork.js/src/utils.js'

    Checks if path is a direct property. (Courtesy of lodash)

    Params:

    NameTypeAttributeDescription
    object Object

    The object to query.
    path Array | string

    The path to check.

    Return:

    boolean

    Returns true if path is a direct property, else false.

    Example:

    
const object = { 'a': { 'b': { 'c': 3 } } };

_.has(object, 'a');
// => true

_.has(object, 'a.b.c');
// => true

_.has(object, ['a', 'b', 'c']);
// => true

    public mixin(target: Class, sources: ...sources<Class>): void source 

    import {mixin} from 'groundwork.js/src/utils.js'

    Mixin properties from a source Class (or sources) into a target Class

    Params:

    NameTypeAttributeDescription
    target Class

    target to merge properties into
    sources ...sources<Class>

    sources to merge properties from

    Return:

    void

    Example:

    class Foo {}
class Bar { hiBar() { return 'Hi from bar!'} }
class Baz { hiBaz() { return 'Hi from baz!'} }

mixin(Foo, Bar, Baz);

const foo = new Foo();
foo.hiBar() // => 'Hi from bar!'
foo.hiBaz() // => 'Hi from baz!'
